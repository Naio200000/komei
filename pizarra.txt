SELECT 
p.id as IdProducto,
p.name as Nombre,
p.descript as Descripcion,
p.id_categoria as Categoria,
p.precio as Precio,
t.name as Tipos,
GROUP_CONCAT(DISTINCT cxp.id_cate_valor separator "|") as Caracterisitcas_Valor,
CONCAT_WS(' ',d.seminario, d.resto) as Disponibilidad,
GROUP_CONCAT(DISTINCT ixp.id_imagen SEPARATOR "|") as imagenes
FROM productos as p
JOIN tipos as t on p.id_tipo = t.id
LEFT JOIN caraval_x_producto as cxp on p.id = cxp.id_producto
JOIN disponibilidad as d on t.id_disponible = d.id
LEFT JOIN imagenes_x_productos as ixp on p.id = ixp.id_producto
Group by IdProducto;
