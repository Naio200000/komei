SELECT
p.id as IdProducto,
p.name as Nombre,
p.descript as Descripcion,
p.precio as Precio,
t.Name as Tipos,
GROUP_CONCAT(CONCAT(c.name, ": ",v.valor) separator " | ") as Caracterisitcas_Valor,
CONCAT_WS(' ',d.seminario, d.resto) as Disponibilidad
FROM productos as p
JOIN tipos as t on p.id_tipo = t.id
LEFT JOIN caraval_x_producto as cxp on p.id = cxp.id_producto
LEFT JOIN valor_x_caracteristica as vxc on cxp.id_cate_valor = vxc.id
JOIN caracteristicas as c on vxc.id_caracteristica = c.id
JOIN valores as v on vxc.id_valor = v.id
JOIN disponibilidad as d on t.id_disponible = d.id
Group by IdProducto;
